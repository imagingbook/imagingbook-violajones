001package imagingbook.violajones.resources.xml;
002
003import java.io.InputStream;
004
005import ij.IJ;
006import imagingbook.lib.util.Enums;
007
008/**
009 * This class gives easy access to the Haar training sets, obtained
010 * from OpenCV. The associated XML files are supposed to be in the SAME 
011 * directory as this class.
012 * @author WB
013 *
014 */
015public enum HaarTrainingSet {
016        eye("haarcascade_eye.xml"),
017        frontalface_default("haarcascade_frontalface_default.xml"),
018        frontalface_alt2("haarcascade_frontalface_alt2.xml")
019        ;
020        
021        private final String filename;
022        
023        private HaarTrainingSet(String filename) {
024                this.filename = filename;
025        }
026        
027        public String getPath() {
028                return this.getClass().getResource(filename).getPath();
029        }
030        
031        public InputStream getStream() {
032                return this.getClass().getResourceAsStream(filename);
033        }
034        
035        //--------------------------------------------
036        
037        public static void listAll() {
038                String[] names = Enums.getEnumNames(HaarTrainingSet.class);
039                
040                for (String name : names) {
041                        IJ.log("Haar set name (enum) = " + name);
042                        HaarTrainingSet set = HaarTrainingSet.valueOf(name);
043                        IJ.log("   filename = " + set.filename);
044                        IJ.log("   full path = " + set.getPath());
045                }
046                
047        }
048
049}
