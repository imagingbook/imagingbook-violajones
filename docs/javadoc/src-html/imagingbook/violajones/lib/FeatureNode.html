001package imagingbook.violajones.lib;
002
003import imagingbook.lib.image.IntegralImage;
004
005
006/**
007 * Represents a Haar feature, usually composed of multiple rectangular
008 * patches (of type {@link FeaturePatch}). Instances of this class make
009 * the nodes of a {@link FeatureTree}.
010 * 
011 * @author WB
012 *
013 */
014public class FeatureNode {
015        
016        protected static final double NO_VALUE = Double.NaN;
017        protected static final int NO_CHILD = -1;
018        protected static final int LEFT = 0;
019        protected static final int RIGHT = 1;
020
021        private final int width, height;
022        private final double threshold;
023        private final double[] values;  // terminal values
024        private final int[] children;   // indexes of child nodes
025        private final FeaturePatch[] patches;
026
027
028        public FeatureNode(int width, int height, double threshold, double valL, int childL, double valR, int childR,
029                        FeaturePatch[] patches) {
030                this.width = width;
031                this.height = height;
032                this.threshold = threshold;
033                this.values = new double[] {valL, valR};
034                this.children = new int[] {childL, childR};
035                this.patches = patches;
036        }
037        
038        public double getValue(int dir) {
039                return values[dir];
040        }
041        
042        public boolean hasValue(int dir) {
043                return !Double.isNaN(values[dir]);
044        }
045        
046        public int getChild(int dir) {
047                return children[dir];
048        }
049        
050        public boolean hasChild(int dir) {
051                return children[dir] != NO_CHILD;
052        }
053        
054        public FeaturePatch[] getPatches() {
055                return patches;
056        }
057
058        /**
059         * Evaluates this feature node by applying its rectangular patches.
060         * @param II input image (as an integral image) 
061         * @param u left position of the candidate subimage
062         * @param v top position of the candidate subimage
063         * @param scale the current scale to be applied
064         * @return
065         */
066        public int eval(IntegralImage II, int u, int v, double scale) {
067                // calculate the area of the subimage:  TODO: w,h never change, make more efficient
068                final int w = (int) Math.round(scale * width);
069                final int h = (int) Math.round(scale * height);
070                
071                // calculate variance (std. deviation) for the complete subimage: TODO: do only once for the subimage?
072                double variance = II.getVariance(u, v, u + w - 1, v + h - 1);
073                double stdDev = (variance > 1) ? Math.sqrt(variance) : 1.0;
074
075                // calculate weighted sum over all patches (rectangles) of this feature
076                double patchSum = 0.0;
077                for (FeaturePatch patch : patches) {
078                        // scale the patch's relative position and size
079                        final int ua = u + (int) Math.round(scale *  patch.x);
080                        final int va = v + (int) Math.round(scale *  patch.y);
081                        final int ub = u + (int) Math.round(scale * (patch.x + patch.width) - 1);
082                        final int vb = v + (int) Math.round(scale * (patch.y + patch.height) - 1);
083                        
084                        // add the patch's block sum and apply the assigned weight (pos/neg)
085                        patchSum = patchSum + II.getBlockSum1(ua, va, ub, vb) * patch.weight;
086                }
087                // normalize the sum of all patches by the effective window area:
088                double patchSumNorm = patchSum / (w * h);
089                        
090                // return direction depending on how the normalized patch sum compares to the threshold
091                return (patchSumNorm < threshold * stdDev) ? LEFT : RIGHT;
092        }
093
094
095        public void print(int featureCnt) {
096                System.out.format("     feature %d: w=%d h=%d valL=%.4f valR=%.4f childL=%d childR=%d th=%.4f patches=%d\n", 
097                                featureCnt, width, height, getValue(LEFT), getValue(RIGHT), getChild(LEFT), getChild(RIGHT), 
098                                threshold, patches.length);
099                int patchCnt = 0;
100                for (FeaturePatch patch : patches) {
101                        System.out.println("           patch " + patchCnt + " " + patch.toString());
102                        patchCnt++;
103                }
104        }
105}
