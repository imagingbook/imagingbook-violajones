019         * Nested class representing parameters and their default values.
020         */
021        public static class Parameters {
022                /** The initial ratio between the window size and the Haar classifier size (default 2). */
023                public double baseScale = 2.00;
024                
025                /** The scale increment of the window size, at each step (default 1.25). */
026                public double scaleStep = 1.25;
027                
028                /** The shift of the window at each sub-step, in terms of percentage of the window size. */
029                public double winShiftFraction = 0.1;
030                
031                /** The minimum number of rectangles needed for the corresponding detection to be kept */
032                public int minNeighbors = 1;
033                
034                /** Flag indicating if Canny pruning should be applied. */
035                public boolean doGradientPruning = false;
036                
037                /** The min. percentage of overlap for merging two regions. */
038                public double minMergeRegionOverlap = 0.2;
039                
040                /** The width of the Gaussian blur applied before gradient calculation. */
041                public double gradientSigma = 2.0;
042                
043                /** The minimum (normalized) gradient magnitude if pruning is on. */
044                public double minGradientMagnitude = 20;
045                
046                /** The maximum (normalized) gradient magnitude if pruning is on. */
047                public double maxGradientMagnitude = 100;
048        }

083        /**
084         * Detects faces in the supplied image.
085         * @param ip the input image
086         * @return the list of rectangles containing searched objects
087         */
088        public List<FaceRegion> findFaces(ByteProcessor ip) {
089                return findFaces(ip.getIntArray());
090        }

092        public List<FaceRegion> findFaces(int[][] I) {
093                List<FaceRegion> facelist = new LinkedList<FaceRegion>();
094                final int imgW = I.length;
095                final int imgH = I[0].length;
096        
097                // maximum size of detector is the whole image 
098                double maxScale = Math.min(
099                                (double) imgW / cascade.getWidth(),
100                                (double) imgH / cascade.getHeight());
101                
102                //IJ.log("maxScale = " + maxScale);
103
104                IntegralImage II = new IntegralImage(I);
105
106                // precalculate Canny edges if option is on
107                //int[][] canny = null;
108                IntegralImage canny = null;
109                if (params.doGradientPruning) {
110                        //canny = CannyPruner.getIntegralCanny(I);
111                        canny = Gradient.makeGradientIntegral(I, params.gradientSigma);
112                }
113
114                // apply the detector at each scale:
115                double scale = params.baseScale;
116                while (scale < maxScale) {
117                        //IJ.log("scale = " + scale);
118                        // calculate the sliding step of the window in each direction:
119                        final int uStep = (int) Math.round(scale * cascade.getWidth() * params.winShiftFraction);
120                        final int vStep = (int) Math.round(scale * cascade.getHeight() * params.winShiftFraction);
121                        final int curW =  (int) Math.round(scale * cascade.getWidth());
122                        final int curH =  (int) Math.round(scale * cascade.getHeight());
123        
124                        // check each position of the window on the image:
125                        for (int v = 0; v < imgH - curH; v += vStep) {
126                                for (int u = 0; u < imgW - curW; u += uStep) {
127                                        /*
128                                         * If Canny pruning is on, compute the edge density of the
129                                         * zone. If it is too low, the object should not be there so
130                                         * skip the region.
131                                         */
132                                        if (params.doGradientPruning) {
133                                                //int edges_density = canny[u + curW][v + curH] + canny[u][v] - canny[u][v + curH] - canny[u + curW][v];
134                                                double gradientMag = canny.getBlockSum1(u, v, u + curW - 1, v + curH - 1);
135                                                double d = gradientMag / (curW * curH);
136                                                if (d < params.minGradientMagnitude || d > params.maxGradientMagnitude)
137                                                        continue;       // skip this region
138                                        }
139                                        
140                                        // apply each detector stage to the current window, reject if one stage fails
141                                        boolean pass = true;
142                                        for (Stage stage : cascade.getStages()) {
143                                                if (!stage.pass(II, u, v, scale)) {
144                                                        pass = false;
145                                                        break;
146                                                }
147                                        }
148                                        // if current window passed all stages, add to the results
149                                        if (pass) {
150                                                facelist.add(new FaceRegion(u, v, curW, curH));
151                                        }
152                                }
153                        }
154                        scale = scale * params.scaleStep;
155                }
156
157                return merge(facelist.toArray(new FaceRegion[0]),
158                                params.minNeighbors, params.minMergeRegionOverlap);
159        }

164        /**
165         * Merge the raw detections resulting from the detection step to avoid
166         * multiple detections of the same object. A threshold on the minimum
167         * numbers of rectangles that need to be merged for the resulting detection
168         * to be kept can be given, to lower the rate of false detections. Two
169         * rectangles need to be merged if they overlap enough.
170         * TODO: CHECK!! Make rectangles immutable!
171         * 
172         * @param facesInit the raw detections returned by the detection algorithm
173         * @param min_neighbors the minimum number of rectangles needed for the corresponding detection to be kept
174         * @param minOverlap the minimum overlap required for merging two faces
175         * @return The merged rectangular detections.
176         */
177        protected List<FaceRegion> merge(FaceRegion[] facesInit, int min_neighbors, double minOverlap) {
178                List<FaceRegion> facesMerged = new LinkedList<FaceRegion>();
179                int[] result = new int[facesInit.length];
180                int nb_classes = 0;
181                for (int i = 0; i < facesInit.length; i++) {
182                        boolean found = false;
183                        for (int j = 0; j < i; j++) {
184                                if (facesInit[j].overlaps(facesInit[i], minOverlap)) {
185                                        found = true;
186                                        result[i] = result[j];
187                                }
188                        }
189                        if (!found) {
190                                result[i] = nb_classes;
191                                nb_classes++;
192                        }
193                }
194                int[] neighbors = new int[nb_classes];
195                FaceRegion[] facesTmp = new FaceRegion[nb_classes];
196                
197                for (int i = 0; i < nb_classes; i++) {
198                        neighbors[i] = 0;
199                        facesTmp[i] = new FaceRegion(0, 0, 0, 0);       // TODO: make more efficient/polish!
200                }
201                
202                for (int i = 0; i < facesInit.length; i++) {
203                        neighbors[result[i]]++;
204                        int x = facesTmp[result[i]].x + facesInit[i].x;
205                        int y = facesTmp[result[i]].y + facesInit[i].y;
206                        int w = facesTmp[result[i]].height + facesInit[i].height;
207                        int h = facesTmp[result[i]].width + facesInit[i].width;
208                        facesTmp[result[i]] = new FaceRegion(x, y, w, h);
209                }
210                
211                for (int i = 0; i < nb_classes; i++) {
212                        int n = neighbors[i];
213                        if (n >= min_neighbors) {
214                                int x = (facesTmp[i].x * 2 + n) / (2 * n);
215                                int y = (facesTmp[i].y * 2 + n) / (2 * n);
216                                int w = (facesTmp[i].width * 2 + n) / (2 * n);
217                                int h = (facesTmp[i].height * 2 + n) / (2 * n);
218                                facesMerged.add(new FaceRegion(x, y, w, h));
219                        }
220                }
221                
222                return facesMerged;
223        }
