001package imagingbook.violajones.lib.util;
002
003import java.awt.Color;
004
005import ij.ImagePlus;
006import imagingbook.lib.color.RandomColorGenerator;
007
008/**
009 * Helper class for visualizing the detection process.
010 * @author WB
011 *
012 */
013public class Visualizer {
014        
015        private static RandomColorGenerator rcg = new RandomColorGenerator();
016        
017        public static ImagePlus im = null;
018        public static boolean on = false;
019        
020        private static Color drawColor = Color.black;
021        private static Color fillColor = Color.white;
022        
023        public static void setDrawColor(Color col) {
024                drawColor = col;
025                if (im != null) {
026                        im.getProcessor().setColor(col);
027                }
028        }
029        
030        public static void clear() {
031                if (im != null) {
032                        im.getProcessor().setColor(fillColor);
033                        im.getProcessor().fill();
034                        im.getProcessor().setColor(drawColor);
035                }
036        }
037        
038        public static void refresh() {
039                if (im != null) {
040                        im.updateAndDraw();
041                }
042        }
043                
044        public static Color nextColor() {
045                return rcg.nextColor();
046        }
047        
048        private static int enableCtr =-1;
049        
050        public static void enable() {
051                enableCtr = 0;
052        }
053        
054        public static void disable() {
055                enableCtr = 1;
056        }
057        
058        public static boolean isEnabled() {
059                return im != null && enableCtr == 0;
060        }
061        
062        public static void suspend() {
063                enableCtr = enableCtr + 1;
064        }
065        
066        public static void unsuspend() {
067                enableCtr = Math.max(0, enableCtr - 1);
068        }
069        
070        public static int getEnableCtr() {
071                return enableCtr;
072        }
073
074}
