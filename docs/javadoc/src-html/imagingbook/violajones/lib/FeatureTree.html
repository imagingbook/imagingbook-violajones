001package imagingbook.violajones.lib;
002
003import imagingbook.lib.image.IntegralImage;
004
005/**
006 * A binary tree for detection. At each node of the tree, a test (feature) is
007 * done on the window. Depending on the result, either the left or right child
008 * of the node is selected and so on. When the process reaches a leaf, the
009 * corresponding value is returned.
010 */
011public class FeatureTree {
012        
013        //private final List<FeatureNode> nodes;
014        private final FeatureNode[] nodes;
015
016        protected FeatureTree(FeatureNode[] nodes) {
017                this.nodes = nodes;
018        }
019        
020        public FeatureNode[] getFeatures() {
021                return nodes;
022        }
023        
024        // evaluate the feature tree
025        protected double eval(IntegralImage II, int u, int v, double scale) {
026                FeatureNode feature = nodes[0]; // start at root of this tree
027                while (true) {
028                        // evaluate the current feature node to decide where to proceed:
029                        int dir = feature.eval(II, u, v, scale);
030                        // if the current node has a value for 'dir', return it:
031                        if (feature.hasValue(dir)) {
032                                return feature.getValue(dir);
033                        }
034                        // otherwise continue with the child node for 'dir'
035                        else {
036                                feature = nodes[feature.getChild(dir)];
037                        }
038                }
039        }
040
041
042        public void print() {
043                int featureCnt = 0;
044                for (FeatureNode node : nodes) {
045                        node.print(featureCnt);
046                        featureCnt++;
047                }
048        }
049
050}
