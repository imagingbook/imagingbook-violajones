021/**
022 * This class represents a trained Haar cascade classifier. Training results were obtained
023 * from OpenCV's 'opencv_haartraining' application (now obsolete, see 
024 * http://docs.opencv.org/doc/user_guide/ug_traincascade.html), available as XML files.
025 * Currently only the old-type XML format is supported.

050         * Creates a Haar cascade from the specification given in a
051         * XML file.
052         * 
053         * @param xmlFilename path to the XML file
054         * @return a new Haar cascade object
055         */
056        public static HaarCascadeDescriptor fromFileName(String xmlFilename) {
057                File file = new File(xmlFilename);
058                String name = file.getName();
059                InputStream strm = HaarTrainingSet.class.getResourceAsStream(name);
060                if (strm == null) {
061                        throw new RuntimeException(HaarCascadeDescriptor.class.getSimpleName() + ": could not find resource " + name);
062                }
063                return HaarCascadeDescriptor.fromInputStream(strm);
064        }

066        /**
067         * Creates a Haar cascade from the specification given in a
068         * XML stream.
069         * 
070         * @param xmlStrm input stream providing XML content
071         * @return a new Haar cascade object
072         */
073        public static HaarCascadeDescriptor fromInputStream(InputStream xmlStrm) {
074                HaarCascadeDescriptor hc = null;
075                DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
076                try {
077                        DocumentBuilder builder = factory.newDocumentBuilder();
078                        Document xmlDoc = builder.parse(xmlStrm);
079                        hc = new HaarCascadeDescriptor();
080                        hc.buildFromXmlDoc(xmlDoc);
081                } catch (Exception e) {
082                        e.printStackTrace();
083                }
084                return hc;
085        }

115        /**
116         * The detector consists of stages, each of them telling whether
117         * the considered zone represents the object with probability a bit
118         * greater than 0.5. If a zone passes all stages, it is considered as
119         * representing the object.
120         * 
121         * @param xmlDoc the XML document object
122         * @throws Exception
123         */
124        private void buildFromXmlDoc(Document xmlDoc) {
125                Element root = getFirstChildElement(xmlDoc.getDocumentElement());
126                System.out.println("Root node name = " + root.getNodeName());
127                
128                //TODO: different readers may be needed for other OpenCV training files
129                String type_id = root.getAttribute("type_id");
130                if (!type_id.equals(XML_TYPE_ID1)) {
131                        throw new RuntimeException("XML file of type " + XML_TYPE_ID1 + " expected.");
132                }
133                
134                /* Read the size (in pixels) of the detector. */
135                Scanner scanner = new Scanner(getChildElement(root, "size").getTextContent()); //new2
136                width = scanner.nextInt();
137                height = scanner.nextInt();
138                scanner.close();
139                
140                stages = new LinkedList<Stage>();
141
142                
143                /* Iterate over the stages nodes to read the stages. */
144                Element stagesE = getChildElement(root, "stages");
145                for (Element stageElem : getChildElements(stagesE, "_")) {
146                        // read the stage threshold:
147                        double stageThreshold = Float.parseFloat(getChildElement(stageElem, "stage_threshold").getTextContent());
148                        Stage stage = new Stage(stageThreshold);
149                        
150                        // read all trees of this stage:
151                        Element treesE = getChildElement(stageElem, "trees");
152                        for (Element treeElem : getChildElements(treesE, "_")) {
153                                
154                                // collect the features contained in this tree:
155                                List<FeatureNode> nodes = new LinkedList<>();
156                                for (Element featureElem : getChildElements(treeElem, "_")) {
157                                        double featureThreshold = Double.parseDouble(getChildElement(featureElem, "threshold").getTextContent());
158                                        
159                                        // get the LEFT value OR child (one of them is empty)
160                                        int childL = FeatureNode.NO_CHILD;
161                                        double valL = FeatureNode.NO_VALUE;
162                                        Element el = getChildElement(featureElem, "left_val");
163                                        if (el != null) {
164                                                valL = Double.parseDouble(el.getTextContent());
165                                        } else {
166                                                childL = Integer.parseInt(getChildElement(featureElem, "left_node").getTextContent());
167                                        }
168
169                                        // get the RIGHT value OR child (one of them is empty)
170                                        int childR = FeatureNode.NO_CHILD;
171                                        double valR = FeatureNode.NO_VALUE;
172                                        Element er = getChildElement(featureElem, "right_val");
173                                        if (er != null) {
174                                                valR = Double.parseDouble(er.getTextContent());
175                                        } else {
176                                                childR = Integer.parseInt(getChildElement(featureElem, "right_node").getTextContent());
177                                        }
178                                        
179                                        List<FeaturePatch> patches = new LinkedList<FeaturePatch>();
180                                        
181                                        // read rectangles of this feature and add:
182                                        Element rectsE = getChildElement(getChildElement(featureElem, "feature"), "rects");
183                                        for (Element rectElem : getChildElements(rectsE, "_")) {
184                                                String s = rectElem.getTextContent().trim();
185                                                FeaturePatch r = FeaturePatch.fromString(s);
186                                                patches.add(r);
187                                        }
188                                        
189                                        FeatureNode feature = new FeatureNode(width, height, featureThreshold, valL, childL, valR, childR, 
190                                                                                                                patches.toArray(new FeaturePatch[0]));
191                                        nodes.add(feature);
192                                }
193                                
194                                FeatureTree tree = new FeatureTree(nodes.toArray(new FeatureNode[0]));
195                                stage.addTree(tree);
196                        }
197                        stages.add(stage);
198                }
199        }

203        /**
204         * Returns the first child node that is of type {@link Element}.
205         * @param parent the parent node
206         * @return the first child node of type {@link Element}.
207         */
208        Element getFirstChildElement(Node parent) {
209                Node child = parent.getFirstChild();
210                while (!(child instanceof Element)) {
211                        child = child.getNextSibling();
212                }
213                return (Element) child;
214        }
215        
216        /**
217         * Collects all child elements into a list.
218         * @param parent the parent node
219         * @return a list of child elements, which may be empty
220         */
221        List<Element> getChildElements(Node parent) {
222                List<Element> elements = new LinkedList<>();
223                NodeList children = parent.getChildNodes();
224                for (int i = 0; i < children.getLength(); i++) {
225                        Node child = children.item(i);
226                        if (child instanceof Element) {
227                                elements.add((Element) child);
228                        }
229                }
230                return elements;
231        }
232        
233        /**
234         * Collects all child elements with the given name into a list.
235         * @param parent the parent node
236         * @param name the name of the child element
237         * @return a list of child elements, which may be empty
238         */
239        List<Element> getChildElements(Node parent, String name) {
240                List<Element> elements = new LinkedList<>();
241                NodeList children = parent.getChildNodes();
242                for (int i = 0; i < children.getLength(); i++) {
243                        Node child = children.item(i);
244                        if (child instanceof Element && name.equals(child.getNodeName())) {
245                                elements.add((Element) child);
246                        }
247                }
248                return elements;
249        }
250
251        /**
252         * Searches for the first immediate child element with the given name.
253         * @param parent the parent node
254         * @param name the name of the child element
255         * @return the first matching child element or {@code null} if none exists.
256         */
257        Element getChildElement(Node parent, String name) {
258                for (Node child = parent.getFirstChild(); child != null; child = child.getNextSibling()) {
259                        if (child instanceof Element && name.equals(child.getNodeName()))
260                                return (Element) child;
261                }
262                return null;
263        }

268                System.out.println("Listing of " + this.getClass().getSimpleName());
269                System.out.format("Size = %d x %d\n", width, height);
270                System.out.format("Number of stages = %d\n", stages.size());
271                int scnt = 0;
272                for (Stage stage : stages) {
273                        scnt++;
274                        System.out.format("******* Stage %d *********\n", scnt);
275                        stage.print();
276                }
277        }
