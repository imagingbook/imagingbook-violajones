007import java.util.List;

010import org.jdom2.Document;

014/**
015 * This class represents a trained Haar cascade classifier. Training results were obtained
016 * from OpenCV's 'opencv_haartraining' application (now obsolete, see 
017 * http://docs.opencv.org/doc/user_guide/ug_traincascade.html), available as XML files.
018 * Currently only the old-type XML format is supported.
019 * TODO: Write a reader for the newer format (produced by 'opencv_traincascade').
020 * 
021 * @author WB
022 *
023 */
024public class HaarCascadeDescriptor {
025        
026        static final String XML_TYPE_ID1 = "opencv-haar-classifier";            // OpenCV "old style"
027        static final String XML_TYPE_ID2 = "opencv-cascade-classifier";         // OpenCV new style?
028        
029        private int width = 0;
030        private int height = 0;
031        private List<Stage> stages = null;
032        
033        // --- constructors ------------------------------
034        
035        // we dont't want to use this from outside:
036        private HaarCascadeDescriptor() {       
037        }
038        
039        // --- static factory methods -------------------
040        
041        /**
042         * Creates a Haar cascade from the specification given in a
043         * XML file.
044         * 
045         * @param xmlFilename path to the XML file
046         * @return a new Haar cascade object
047         */
048        public static HaarCascadeDescriptor createFrom(String xmlFilename) {
049                HaarCascadeDescriptor hc = null;
050                try {
051                        Document xmlDoc = new SAXBuilder().build(new File(xmlFilename));
052                        hc = new HaarCascadeDescriptor();
053                        hc.buildFrom(xmlDoc);
054                } catch (Exception e) {
055                        e.printStackTrace();
056                }
057                return hc;
058        }
059        
060        /**
061         * Creates a Haar cascade from the specification given in a
062         * XML stream.
063         * 
064         * @param xmlStrm input stream providing XML content
065         * @return a new Haar cascade object
066         */
067        public static HaarCascadeDescriptor createFrom(InputStream xmlStrm) {
068                HaarCascadeDescriptor hc = null;
069                try {
070                        Document xmlDoc = new SAXBuilder().build(xmlStrm);
071                        hc = new HaarCascadeDescriptor();
072                        hc.buildFrom(xmlDoc);
073                } catch (Exception e) {
074                        e.printStackTrace();
075                }
076                return hc;
077        }
078
079        // --- getters and setters ------------------------
080        
081        /**
082         * Returns the width of this Haar cascade.
083         * @return the width of this Haar cascade (in pixels units)
084         */
085        public int getWidth() {
086                return width;
087        }
088        
089        /**
090         * Returns the height of this Haar cascade.
091         * @return the height of this Haar cascade (in pixels units)
092         */
093        public int getHeight() {
094                return height;
095        }
096        
097        /**
098         * Returns the number of stages in this Haar cascade.
099         * @return the number of stages this Haar cascade
100         */
101        public List<Stage> getStages() {
102                return stages;
103        }
104
105        // ------------------------------------------------
106        
107        // See jdom2 API documentation: http://www.jdom.org/docs/apidocs/
108        
109        private void buildFrom(Document xmlDoc) throws Exception {
110                /*
111                 * The detector is constituted by stages, each of them telling whether
112                 * the considered zone represents the object with probability a bit
113                 * greater than 0.5. If a zone passes all stages, it is considered as
114                 * representing the object.
115                 */
116                
117                Element root = xmlDoc.getRootElement().getChildren().get(0);
118                
119                // added by WB, TODO: different readers may be needed for other OpenCV training files
120                String type_id = root.getAttributeValue("type_id");
121                //IJ.log("root attribute =" + type_id);
122
123                if (!type_id.equals(XML_TYPE_ID1)) {
124                        throw new Exception("XML file of type " + XML_TYPE_ID1 + " expected.");
125                }
126                
127                /* Read the size (in pixels) of the detector. */
128                Scanner scanner = new Scanner(root.getChild("size").getText());
129                width = scanner.nextInt();
130                height = scanner.nextInt();
131                scanner.close();
132                
133                stages = new LinkedList<Stage>();
134
135                /* Iterate over the stages nodes to read the stages. */
136                for (Element stageElem : root.getChild("stages").getChildren("_")) {
137                        // read the stage threshold:
138                        double stageThreshold = Float.parseFloat(stageElem.getChild("stage_threshold").getText());
139                        Stage stage = new Stage(stageThreshold);
140                        
141                        // read all trees of this stage:
142                        for (Element treeElem : stageElem.getChild("trees").getChildren("_")) {
143                                
144                                // collect the features contained in this tree:
145                                List<FeatureNode> nodes = new LinkedList<>();
146                                for (Element featureElem : treeElem.getChildren("_")) {
147                                        double featureThreshold = Double.parseDouble(featureElem.getChild("threshold").getText());
148                                        
149                                        // get the LEFT value OR child (one of them is empty)
150                                        int childL = FeatureNode.NO_CHILD;
151                                        double valL = FeatureNode.NO_VALUE;
152                                        Element el = featureElem.getChild("left_val");
153                                        if (el != null) {
154                                                valL = Double.parseDouble(el.getText());
155                                        } else {
156                                                childL = Integer.parseInt(featureElem.getChild("left_node").getText());
157                                        }
158
159                                        // get the RIGHT value OR child (one of them is empty)
160                                        int childR = FeatureNode.NO_CHILD;
161                                        double valR = FeatureNode.NO_VALUE;
162                                        Element er = featureElem.getChild("right_val");
163                                        if (er != null) {
164                                                valR = Double.parseDouble(er.getText());
165                                        } else {
166                                                childR = Integer.parseInt(featureElem.getChild("right_node").getText());
167                                        }
168                                        
169                                        List<FeaturePatch> patches = new LinkedList<FeaturePatch>();
170                                        
171                                        // read rectangles of this feature and add:
172                                        for (Element rectElem : featureElem.getChild("feature").getChild("rects").getChildren("_")) {
173                                                String s = rectElem.getText().trim();
174                                                FeaturePatch r = FeaturePatch.fromString(s);
175                                                patches.add(r);
176                                        }
177                                        
178                                        FeatureNode feature = new FeatureNode(width, height, featureThreshold, valL, childL, valR, childR, 
179                                                                                                                patches.toArray(new FeaturePatch[0]));
180                                        nodes.add(feature);
181                                }
182                                
183                                FeatureTree tree = new FeatureTree(nodes.toArray(new FeatureNode[0]));
184                                stage.addTree(tree);
185                        }
186                        stages.add(stage);
187                }
188        }
189        
190        @Deprecated
191        public void print() {
192                System.out.println("Listing of " + this.getClass().getSimpleName());
193                System.out.format("Size = %d x %d\n", width, height);
194                System.out.format("Number of stages = %d\n", stages.size());
195                int scnt = 0;
196                for (Stage stage : stages) {
197                        scnt++;
198                        System.out.format("******* Stage %d *********\n", scnt);
199                        stage.print();
200                }
201        }
