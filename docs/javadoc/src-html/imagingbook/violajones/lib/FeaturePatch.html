001package imagingbook.violajones.lib;
002
003import java.awt.Rectangle;
004
005/** 
006 * Represents a rectangle, along with a weight. 
007 */
008public class FeaturePatch extends Rectangle {
009        private static final long serialVersionUID = 1L;
010        
011        public final double weight;
012
013        public FeaturePatch(int x, int y, int w, int h, double weight) {
014                super(x, y, w, h);
015                this.weight = weight;
016        }
017
018        public static FeaturePatch fromString(String text) {
019                String[] tab = text.split(" ");
020                FeaturePatch fp = null;
021                try {
022                        int x = java.lang.Integer.parseInt(tab[0]);
023                        int y = java.lang.Integer.parseInt(tab[1]);
024                        int w = java.lang.Integer.parseInt(tab[2]);
025                        int h = java.lang.Integer.parseInt(tab[3]);
026                        double weight = java.lang.Double.parseDouble(tab[4]);
027                        fp = new FeaturePatch(x, y, w, h, weight);
028                } catch (NumberFormatException e) {
029                        System.out.println(FeaturePatch.class.getSimpleName() + ": NumberFormatException");
030                }
031                return fp;
032        }
033        
034        public String toString() {
035                return String.format("x=%d y=%d w=%d h=%d weight=%.2f", 
036                                this.x, this.y, this.width, this.height, this.weight);
037        }
038}
