003import java.awt.Rectangle;

005/**
006 * Merely a rectangle which can check for overlap with another rectangle.
007 * @author WB
008 *
009 */
010public class FaceRegion extends Rectangle {
011        private static final long serialVersionUID = 1L;
012
013        public FaceRegion(int x, int y, int w, int h) {
014                super(x, y, w, h);
015        }
016        
017        // TODO: check!!
018        public boolean overlaps(Rectangle r2, double sizeFac) {
019                final double sizeFac1 = 1 + sizeFac;
020                int distance = (int) (this.width * sizeFac);
021                if (r2.x <= this.x + distance && r2.x >= this.x - distance
022                                && r2.y <= this.y + distance && r2.y >= this.y - distance
023                                && r2.width <= (int) (this.width * sizeFac1)
024                                && (int) (r2.width * sizeFac1) >= this.width) {
025                        return true;
026                }
027                if (this.x >= r2.x && this.x + this.width <= r2.x + r2.width && this.y >= r2.y
028                                && this.y + this.height <= r2.y + r2.height) {
029                        return true;
030                }
031                        
032                return false;
033        }
034
035}
