001package Face_Plugins;
002
003import java.awt.Color;
004import java.util.List;
005
006import ij.IJ;
007import ij.ImagePlus;
008import ij.gui.GenericDialog;
009import ij.plugin.filter.PlugInFilter;
010import ij.process.ImageProcessor;
011import imagingbook.lib.ij.IjUtils;
012import imagingbook.lib.util.Enums;
013import imagingbook.violajones.lib.FaceDetector;
014import imagingbook.violajones.lib.FaceDetector.Parameters;
015import imagingbook.violajones.lib.FaceRegion;
016import imagingbook.violajones.resources.xml.HaarTrainingSet;
017
018
019/**
020 * Face detection ImageJ plugin using 'imagingbook.violajones' implementation
021 * of the Viola-Jones face detection algorithm: Paul Viola and Michael Jones, 
022 * "Robust Real-time Face Detection", International Journal of Computer Vision 
023 * 57.2 (2004), pp. 137â€“154.
024 * 
025 * @author W. Burger
026 * @version 2018/12/09
027 */
028public class Find_Faces implements PlugInFilter {
029        
030        static boolean extractFaceImages = false;
031        static HaarTrainingSet trainingSet = HaarTrainingSet.frontalface_alt2;
032        
033        ImagePlus im = null;
034
035        public int setup(String arg0, ImagePlus im) {
036                this.im = im;
037                return DOES_ALL + NO_CHANGES;
038        }
039        
040        public void run(ImageProcessor ip) {
041                
042                Parameters params = new Parameters();
043                params.baseScale = 1.5;
044                params.scaleStep = 1.05;
045                params.winShiftFraction = 0.1;
046                params.minNeighbors = 1;
047                params.minMergeRegionOverlap = 0.2;
048                
049                params.doGradientPruning = false;
050                params.gradientSigma = 1.0;
051                params.minGradientMagnitude = 5;
052                params.maxGradientMagnitude = 100;
053                
054                if (!setParameters(params)) return;
055                
056                //IJ.log("Using cascade decriptor " + trainingSet.getPath());
057                
058                IjUtils.setRgbConversionWeights(ip);
059                FaceDetector detector = FaceDetector.create(trainingSet.getStream(), params);
060                List<FaceRegion> faces = detector.findFaces(ip.convertToByteProcessor());
061                IJ.log(faces.size() + " faces found!");
062                
063                // show results
064                ImageProcessor cp = ip.convertToColorProcessor();
065                for (FaceRegion r : faces) {
066                        draw(r, cp);
067                }
068                new ImagePlus(this.getClass().getSimpleName() + ": results", cp).show();
069                
070                if (extractFaceImages && !faces.isEmpty()) {
071                        int faceCtr = 0;
072                        for (FaceRegion r : faces) {
073                                ip.setRoi(r.x, r.y, r.width, r.height);
074                                new ImagePlus("Face " + faceCtr, ip.crop()).show();
075                                faceCtr++;
076                        }
077                }
078        }
079        
080        boolean setParameters(Parameters params) {
081                GenericDialog gd = new GenericDialog("Set Face Detector Parameters");
082                gd.addChoice("Haar training set", Enums.getEnumNames(HaarTrainingSet.class), trainingSet.name());
083                gd.addNumericField("baseScale", params.baseScale, 2);
084                gd.addNumericField("scaleStep", params.scaleStep, 2);
085                gd.addNumericField("winShiftFraction", params.winShiftFraction, 2);
086                gd.addNumericField("minNeighbors", params.minNeighbors, 0);
087                gd.addNumericField("minMergeRegionOverlap", params.minMergeRegionOverlap, 2);
088                
089                gd.addCheckbox("doGradientPruning", params.doGradientPruning);
090                
091                gd.addNumericField("gradientSigma", params.gradientSigma, 2);
092                gd.addNumericField("minGradientMagnitude", params.minGradientMagnitude, 0);
093                gd.addNumericField("maxGradientMagnitude", params.maxGradientMagnitude, 2);
094                
095                gd.addCheckbox("extractFaceImages", extractFaceImages);
096                
097                gd.showDialog();
098                if (gd.wasCanceled()) {
099                        return false;
100                }
101                
102                trainingSet = HaarTrainingSet.valueOf(gd.getNextChoice());
103                params.baseScale = gd.getNextNumber();
104                params.scaleStep = gd.getNextNumber();
105                params.winShiftFraction = gd.getNextNumber();
106                params.minNeighbors = (int) gd.getNextNumber();
107                params.minMergeRegionOverlap = gd.getNextNumber();
108                
109                params.doGradientPruning = gd.getNextBoolean();
110                
111                params.gradientSigma = gd.getNextNumber();
112                params.minGradientMagnitude = gd.getNextNumber();
113                params.maxGradientMagnitude = gd.getNextNumber();
114                
115                extractFaceImages = gd.getNextBoolean();
116                return true;
117        }
118
119        private void draw(FaceRegion f, ImageProcessor cp) {
120                cp.setColor(Color.green);
121                cp.drawRect(f.x, f.y, f.width, f.height);
122        }
123
124}
