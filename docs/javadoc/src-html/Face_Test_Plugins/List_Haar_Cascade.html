001package Face_Test_Plugins;
002
003import java.io.File;
004import java.io.InputStream;
005
006import ij.IJ;
007import ij.io.OpenDialog;
008import ij.plugin.PlugIn;
009import imagingbook.lib.ij.IjLogStream;
010import imagingbook.lib.util.ResourceUtils;
011import imagingbook.violajones.lib.HaarCascadeDescriptor;
012import imagingbook.violajones.resources.Data;
013import imagingbook.violajones.resources.xml.HaarTrainingSet;
014
015
016/**
017 * This plugin lists the available OpenCV cascade descriptor files.
018 * Currently only "old style" XML files (of type "opencv-haar-classifier")
019 * can be read.
020 * 
021 * @author WB
022 *
023 */
024public class List_Haar_Cascade implements PlugIn {
025        
026        static {
027                IjLogStream.redirectSystem();
028        }
029
030
031        public void run(String arg) {
032                String xmlRoot = ResourceUtils.getResourcePath(Data.class, "xml").toString();
033                //String xmlRoot = ResourceUtils.getResourcePath(HaarTrainingSet.class, "").toString();
034                OpenDialog od = new OpenDialog("Select XML file", xmlRoot, "");
035                String xmlPath = od.getPath();
036                if (xmlPath == null) return;
037                if (!xmlPath.endsWith(".xml")) {
038                        IJ.error("this is not an XML file!");
039                        return;
040                }
041
042                System.out.println("XML path: " + xmlPath);
043                
044                File file = new File(xmlPath);
045                String name = file.getName();
046                System.out.println("XML file: " + name);
047                                
048//              InputStream strm = ResourceUtils.getResourceStream(Data.class, "xml/haarcascade_frontalface_alt2.xml");
049//              InputStream strm = Data.class.getResourceAsStream("xml/haarcascade_frontalface_alt2.xml");
050//              
051                InputStream strm = HaarTrainingSet.class.getResourceAsStream(name);
052                if (strm == null) {
053                        IJ.log("could not open XML stream for " + xmlPath);
054                        return;
055                }
056
057                IJ.log("Reading XML stream ...");
058                HaarCascadeDescriptor hc = HaarCascadeDescriptor.createFrom(strm);
059                hc.print();
060
061                IJ.log("done.");
062        }
063
064
065
066
067}
