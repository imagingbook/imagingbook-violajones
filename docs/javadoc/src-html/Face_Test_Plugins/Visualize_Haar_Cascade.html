024/**
025 * Visualizes the evaluation sequence for the selected Haar cascade.
026 * The current image is scaled to the size of the cascade shown in
027 * the background. The result is an image stack, with one slice for
028 * each decision step.
029 * 
030 * @author WB
031 *
032 */
033public class Visualize_Haar_Cascade implements PlugInFilter {
034
035        public int setup(String arg0, ImagePlus im) {
036                return DOES_ALL + NO_CHANGES;
037        }
038
039        public void run(ImageProcessor ip) {
040        
041                HaarTrainingSet desc = HaarTrainingSet.frontalface_alt2;
042                InputStream strm = desc.getStream();
043                if (strm == null) {
044                        IJ.log("could not open training set");
045                        return;
046                }
047                
048                HaarCascadeDescriptor cascade = HaarCascadeDescriptor.fromInputStream(strm);
049                int w = cascade.getWidth();
050                int h = cascade.getHeight();
051                
052                List<Stage> stages = cascade.getStages();
053                
054                // create and populate the stack
055                ImageStack stack = new ImageStack(w, h);
056                ImageProcessor sp1 = ip.resize(w, h);
057                
058                int stageCtr = 0;
059                for (Stage stage : stages) {
060                        int treeCtr = 0;
061                        for (FeatureTree tree : stage.getTrees()) {
062                                int featureCtr = 0;
063                                for (@SuppressWarnings("unused") 
064                                                FeatureNode feature : tree.getFeatures()) {
065                                        ImageProcessor sp = sp1; //sp1.duplicate();
066//                                      sp.setColor(Color.gray);
067//                                      sp.fill();
068//                                      draw(sp, feature);
069                                        stack.addSlice("Stage=" + stageCtr + " Tree=" + treeCtr + " Feat=" + featureCtr, sp);
070                                        featureCtr++;
071                                }
072                                treeCtr++;
073                        }
074                        stageCtr++;
075                }
076
077                // add overlay graphics
078                Overlay oly = new Overlay();
079                int sliceNo = 1;
080                for (Stage stage : stages) {
081                        for (FeatureTree tree : stage.getTrees()) {
082                                for (FeatureNode feature : tree.getFeatures()) {
083                                        for (FeaturePatch r : feature.getPatches()) {
084                                                Color col = (r.weight > 0) ? Color.green : Color.red;
085                                                Roi box = new Roi(r.x, r.y, r.width, r.height);
086                                                box.setStrokeColor(col);
087                                                box.setPosition(sliceNo);
088                                                Roi diag = new Line(r.x, r.y, r.x + r.width, r.y + r.height);
089                                                diag.setStrokeColor(col);
090                                                diag.setPosition(sliceNo);
091                                                oly.add(box);
092                                                oly.add(diag);
093                                        }
094                                        sliceNo++;
095                                }
096                        }
097                }
098                
099                ImagePlus stackim = new ImagePlus("Cascade", stack);
100                stackim.setOverlay(oly);
101                stackim.show();
102                
103        }
104        
105        List<Roi> getRectangles(FeatureNode feature) {
106                List<Roi> rectRois = new ArrayList<Roi>();
107                for (FeaturePatch r : feature.getPatches()) {
108                        rectRois.add(new Roi(r.x, r.y, r.width, r.height));
109                }
110                return rectRois;
111        }
112
113        @SuppressWarnings("unused")
114        private void draw(ImageProcessor sp, FeatureNode feature) {
115                for (FeaturePatch r : feature.getPatches()) {
116                        Color col = (r.weight > 0) ? Color.green : Color.red;
117                        sp.setColor(col);
118                        for (int u = 0; u < r.width; u++) {
119                                for (int v = 0; v < r.height; v++) {
120                                        sp.drawPixel(u + r.x, v + r.y);
121                                }
122                        }
123                        //sp.drawRect(r.x, r.y, r.width, r.height);
124                        
125                }
126                
127        }
128
129        @SuppressWarnings("unused")
130        private void draw(FaceRegion f, ImageProcessor cp) {
131                cp.setColor(Color.green);
132                cp.drawRect(f.x, f.y, f.width, f.height);
133        }
